package com.techmasters.continued.placement;
import java.util.Scanner;

public class Perfectsquare {

   public static boolean isPerfectSquare(int n) {
       if(n<0) {
           return false;
       }
       int sqrt=(int)Math.sqrt(n);//double 49.0 ==> int 7
       return(sqrt*sqrt==n);
   }
   
   public static boolean isperfectsqr(int num) {
       if(num<0) return false;
       if(num==0 || num ==1) return true;
       
       for(int i=1; i*i<=num; i++) {
           if(i*i==num) return true;
       }
       return false;
    }

   public static void main(String[] args) {
       // TODO Auto-generated method stub
       Scanner sc=new Scanner(System.in);
       System.out.println("Enter the number to check if perfect sqaure");
       int n=sc.nextInt();
       if(isPerfectSquare(n)) {
           System.out.println(n +" is a perfect sqaure number");
       }else {
           System.out.println(n + " is not a perfect square number");
       }
       System.out.println("Enter the number to check if perfect sqaure");
       int num=sc.nextInt();
       if(isperfectsqr(num)) {
           System.out.println(num + " is a perfect sqaure number");
       }else {
           System.out.println(num + " is not a perfect sqaure number");
       }
   }
}
